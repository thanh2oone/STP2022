// https://www.spoj.com/problems/STPAR/

#include<iostream>
using namespace std;

int a[1000];
//int carWait[1000];

int st[1000];
int sizeStack = 0;

void push(int value) {
	sizeStack++;
	st[sizeStack] = value;
}

int top() {
	return st[sizeStack];
}

void pop() {
	sizeStack--;
}

bool empty() {
	if (sizeStack == 0) return true;
	return false;
}

int main() {
	freopen("Input.txt", "r", stdin);
	ios_base::sync_with_stdio;
	cin.tie(NULL);
	cout.tie(NULL);

	while (true) {
		int n; cin >> n;
		if (n == 0) break;

		for (int i = 1; i <= n; i++) {
			cin >> a[i];
		}

		/*for (int i = 1; i <= n; i++) {
			carWait[i] = 0;
		}*/

		while (!empty()) {
			pop();
		}

		int check = 1; // Đếm số lượng và kiểm tra xe đúng vị trí
		//int sizeQueue = 0;
		for (int i = 1; i <= n; i++) {
			if (a[i] != check) {
				// Cho xe sai vị trí vào hàng đợi
				//sizeQueue++;
				//carWait[sizeQueue] = a[i];
				push(a[i]);
			}

			// Kiểm tra xe đúng vị trí
			if (a[i] == check) {
				check++; // Kiểm tra xe tiếp theo
			}

			if (!empty()) {
				//if (sizeQueue != 0) {
					//while (carWait[sizeQueue] == check) {
				int temp = top();
				while (temp == check) {
					//carWait[sizeQueue] = 0;
					//sizeQueue--;
					pop();
					check++; // Kiểm tra xe tiếp theo
					if (!empty()) temp = top();
				//}
				}
			}

		}

		if (check - 1== n) cout << "yes" << endl;
		else cout << "no" << endl;
	}
	return 0;
}
